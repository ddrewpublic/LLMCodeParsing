int sushu(int b) {
  // Declare a flag variable to keep track of whether the number is prime or not
  int flag = 1, i;
  
  // If the number is 2, it is considered prime, so return 1
  if (b == 2) return 1;

  if (b % 2 == 0) return 0;
  
  // Loop through all numbers from 2 to half of the input number
  for (i = 2; i <= b / 2; i++) {
    // If the number is divisible by any of these numbers, it is not prime, so break the loop
    if (b % i == 0) {
      flag = 0;
      break;
    }
  }
  
  // If the flag is still 1, the number is prime, so return 1
  if (flag == 0) return 0;
  else return 1;
}

  int i, total = 1;

      if (a % i == 0 && a / i >= i) {

        total = total + number(a / i, i);

  return total;
}
