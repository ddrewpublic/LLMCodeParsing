int im(int n, int max)
{
    // initialize a variable to keep track of the sum of divisors
    int s = 0;
    // initialize a loop variable to iterate through numbers up to max
    int i;
    // if the input number is 1, return 1 (since 1 has only 1 proper divisor)
    if (n == 1)
        return 1;
    // iterate through numbers from max down to 2
    for (i = max; i > 1; i--)
    {
        // if the current number is a divisor of the input number, recursively add its sum to s
        if (n % i == 0)
            s += im(n / i, i);
    }
    // return the calculated sum of divisors
    return s;
}
