```cpp
int sum;  // Declaring a global variable sum of type integer

void calculate(int, int);  // Declaring a function called calculate with two integer parameters and no return type

int main()  // The main function which is the entry point of the program
{
    int n;  // Declaring a variable n of type integer
    cin >> n;  // Reading input from the user and storing it in variable n
    for (int i = 0; i < n;i++)  // Starting a for loop which will run n times
    {
        int m;  // Declaring a variable m of type integer
        sum  = 1;  // Initializing the global variable sum to 1
        cin >> m;  // Reading input from the user and storing it in variable m
        calculate(m, 2);  // Calling the function calculate with parameters m and 2
        cout << sum << endl;  // Printing the value of sum followed by a newline
    }
    return 0;  // Returning 0 to indicate that the program has executed successfully
}
void calculate(int m,int begin)  // Defining the function calculate
{
    int i;  // Declaring a variable i of type integer
    int n = sqrt((double)m);  // Calculating the square root of m and storing it in variable n
    for ( i = begin; i <= n; i++ )  // Starting a for loop which will run until i is greater than n
    {
        if ( m % i == 0)  // Checking if m is divisible by i
        {
            sum++;  // Incrementing the global variable sum by 1
            calculate(m / i,i);  // Recursively calling the function calculate with parameters m/i and i
        }
    }
}
```

**Part II**

This program calculates the number of divisors of a given number. It does this by recursively adding to the total count of divisors for each pair of numbers, where one number is the divisor and the other is the quotient.
