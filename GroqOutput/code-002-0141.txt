// Define a struct called book with two members: an integer "num" and a character array "str" of size 26.
struct book
{
	int num;
	char str[26];
}bk[1000];

void main()
{
	// Declare variables: m for the number of books, i, j, k, and s for loops, and n for the maximum frequency.
	int m,i,j,k,s,n=0;
	// Define an array of characters a with all uppercase letters.
	char a[26]={'A','B','C','D','E','F','G','H','I','J','K','L','M','N','O','P','R','S','T','U','V','W','X','Y','Z'};
	// Create an array b with 26 elements, all initialized to 0.
	int b[26]={0};
	// Read the number of books m from the user.
	scanf("%d",&m);
    // Loop through each book and read its number and title.
	for(i=0;i<m;i++)
		scanf("%d %s",&bk[i].num,bk[i].str);
