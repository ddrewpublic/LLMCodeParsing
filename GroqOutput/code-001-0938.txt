int geshu;// variable to store the result of the yinshu() function

void yinshu(int num, int a)// function to find the number of prime factors of a given number
{
    int i; // loop variable
    if (num == 1) geshu++;// increment geshu if num is 1 (since 1 is not a prime number)
    else 
    {
        for (i = a; i <= num; i++)
        {
            // check if num is divisible by i (i.e., num % i == 0)
            if (num % i == 0)
            {
                // if num is divisible by i, recursively call yinshu() with num/i and i as arguments
                yinshu(num / i, i);
            }
        }
    }
}

int main()
{
    int shu, j, num1[10000];// array to store numbers, shu to store the number of inputs, j to store the input number index
    cin >> shu; // read the number of inputs
    for (j = 0; j < shu; j++)
    {
        geshu = 0; // reset geshu for each input
        cin >> num1[j]; // read the input number
        yinshu(num1[j], 2); // find the number of prime factors of the input number starting from 2
        cout << geshu << endl; // print the result
    }
    cin.get(); cin.get(); cin.get(); cin.get(); cin.get(); // wait for user input
    return 0;
}
