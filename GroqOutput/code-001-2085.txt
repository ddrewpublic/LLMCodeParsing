{
	// This function returns the square root of a given integer y
    int i;
    for(i=0;;i++)
    {
        // This loop increments i until i*i is greater than y
        if(i*i>y)
            break;
    }
    // The function returns i-1, which is the square root of y
    return i-1;
}
int f(int x,int b)
{
    // This function computes the sum of all prime numbers less than or equal to x
    int p,a[100],t=0,k=0;
    p=g(x);
    // The function calculates the greatest prime number less than or equal to x
    if(x==2)
        return 0;
    else
    {
        // The function finds all prime numbers less than or equal to p
        for(int i=b;i<=p;i++)
        {
            // This loop checks if x is divisible by i
            if(x%i==0)
            {
                // If x is divisible by i, i is a prime number and is added to the array a
                a[t]=i;
                t++;
            }
        }
        // If no prime numbers are found, the function returns 0
        if(t==0)
            return 0;
        else
        {
            // The function calculates the sum of prime numbers by recursively calling itself
            for(int i=0;i<t;i++)
                k=k+f(x/a[i],a[i])+1;
            // The function returns the sum of prime numbers
            return k;
        }
    }
}
