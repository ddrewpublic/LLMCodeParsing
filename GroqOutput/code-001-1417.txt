```c++
int a=1; // Initialize the variable a to 1
int yinzi(int j,int n) // Function yinzi takes two integer parameters j and n
{   
    for(int i=j;i<n;i++) // Loop from j to n-1
	{
		if(n%i==0&&n!=i&&n/i>=i) // Check if n is divisible by i, not equal to i, and n divided by i is greater than or equal to i
			{  
			    a++; // Increment the value of a
				yinzi(i,n/i); // Recursively call the yinzi function with parameters i and n/i
			}
		
	}
	return 0; // Return 0

			
}
int main()
{
	int n,s; // Declare integer variables n and s
	cin>>s; // Read an integer from the input and store it in s
	for(int j=0;j<s;j++) // Loop s times
	{
		cin>>n; // Read an integer from the input and store it in n
		for(int i=2;i<=n;i++) // Loop from 2 to n
		{
			if(n%i==0&&n/i>=i) // Check if n is divisible by i and n divided by i is greater than or equal to i
			{
				a++; // Increment the value of a
		         yinzi(i,n/i); // Recursively call the yinzi function with parameters i and n/i
			}
		}
	    cout<<a<<endl; // Print the value of a followed by a newline
	    a=1; // Reset the value of a to 1
	}
	return 0;
}
```
