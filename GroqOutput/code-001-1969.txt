int sushu(int a) {
    // Initialize variables
    int b = 2; // starting number for primality test
    int c, d = 0, k;
    
    // Loop until 'b' is greater than 'a'
    while (b < a) {
        // Calculate remainder of 'a' divided by 'b'
        c = a % b;
        
        // If the remainder is not 0, increment 'b' and 'd'
        if (c != 0) {
            b = b + 1;
            d = d + 1;
        }
        // If the remainder is 0, break the loop
        else
            break;
    }
    
    // If 'd' is equal to 'a-2', 'a' is prime
    if(d == (a-2))
        k = 1;
    else
        k = 0;
    
    // Return whether 'a' is prime or not
    return k;
}

int zheng(int n, int i) {
    // Initialize variables
    int sum = 1, a;
    
    // Loop until 'a' is less than or equal to 'n/2'
    for(a = i; a < n/2; a++) {
        // If 'n' is divisible by 'a'
        if(n % a == 0) {
            // If 'a' is less than or equal to 'n/a', recursively call 'zheng'
            if(a <= n/a) {
                sum = sum + zheng(n/a, a);
            }
        }
    }
    
    // Return the sum
    return sum;
}

int main () {
    // Initialize variables
    int n, i, j = 0, k = 0, b[10000] = {0}, e[40000] = {0};
    
    // Read the number 'n'
    scanf("%d", &n);
    
    // Read 'n' numbers
    for(i = 0; i < n; i++) {
        // Read the 'i'th number
        scanf("%d", &e[i]);
    }
    
    // Loop through the numbers
    for(i = 0; i < n; i++) {
        // Check if the number is prime
        if(sushu(e[i]) == 0) {
            // Calculate the sum of proper divisors of the number
            b[i] = zheng(e[i], 2);
        }
        else {
            // The number is prime, set the sum to 1
            b[i] = 1;
        }
    }
    
    // Print the results
    printf("%d", b[0]);
    for(i = 1; i < n; i++)
        printf("\n%d", b[i]);
    
    return 0;
}

The code is designed to check for prime numbers and calculate the sum of their proper divisors. The main function reads in a number 'n' and 'n' numbers from the input. For each number, it checks if it is prime using the 'sushu' function, and if so, calculates the sum of its proper divisors using the 'zheng' function.
