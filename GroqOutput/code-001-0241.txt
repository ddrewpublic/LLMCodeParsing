```cpp
int means(int,int);
int main()
{
    int n,i=0;  // Declare variables n and i, n will hold the number of inputs and i will be used as a counter
    cin>>n;  // Take the number of inputs from the user
    while(i<n)
    {
        int a;  // Declare variable a to hold the current input
        cin>>a;  // Take the current input from the user
        cout<<means(a,a)<<endl;  // Calculate the sum of proper divisors of a and print it
        i++;  // Increment the counter
    }
    return 0;  // Return 0 to indicate successful execution
}

int means(int a,int b)
{
    int i,sum=0;  // Declare variables i and sum, sum will hold the sum of proper divisors
    if(a!=1)  // If a is not equal to 1
    {
        for(i=b;i>=2;i--)  // Start from b and go down to 2
        {
            if(a%i==0)  // If a is divisible by i
            {
                sum=sum+means(a/i,i);  // Recursively calculate the sum of proper divisors of a/i
            }
        }
        return sum;  // Return the sum of proper divisors of a
    }
    else
    {
        return 1;  // If a is 1, return 1
    }
}
```
