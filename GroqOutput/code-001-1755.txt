int fenjie(int a,int max){//a: input number, max: maximum divisor
    int i = 0;
    int sum = 0;//initialize sum to 0
    if (a == 1) return 1;//base case: if a is 1, return 1
    for (i = max;i >= 2;i--){
        if ((a % i) == 0) sum += fenjie(a/i,i);//recurse for all divisors
    }
    return sum;//return the sum of divisors
}

int main() {
    int k = 0;
    int n = 0;
    int num = 0;
    cin >> n;//read the number of inputs
    for (k = 1;k <= n;k++){
        cin >> num;//read a number
        cout << fenjie(num,num);//compute and print the divisor sum
        if (k < n) cout << endl;//print a newline if not the last input
    }
    return 0;
}
