int f(int a,int b)
{
    // If a and b are equal, return 1
    if(a==b)return 1;
    // If b is greater than a, return 0
    if(b>a)return 0;
    // If a is not divisible by b, recursively call f with b+1
    if(a%b!=0)return f(a,b+1);
    // If a is divisible by b, recursively call f with a/b and b
    if(a%b==0)return f(a,b+1)+f(a/b,b);
}
