**Part I: Code**
```c
int num=0; // Initialize a variable num to 0

void count(int m,int j) // Define a function count
{
    int i; // Declare a variable i
    if(m==1) num++; // If m is 1, increment the num variable
    for(i=j;i<=m;i++) // Loop from j to m
    {
        if(m%i==0) {count(m/i,i);} // If m is divisible by i, recursively call count with m/i and i
        else continue; // If not, skip to the next iteration
    }
}

int main() // Define the main function
{
    int n,i; // Declare variables n and i
    cin>>n; // Read an integer from the input
    for(i=1;i<=n;i++) // Loop n times
    {
        int m; // Declare a variable m
        cin>>m; // Read an integer from the input
        num=0; // Reset the num variable to 0
        count(m,2); // Call the count function with m and 2
        cout<<num<<endl; // Print the value of num followed by a newline
    }
    return 0; // Return 0 to indicate the program has ended successfully
}
```
