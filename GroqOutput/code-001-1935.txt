// Function to implement the calculation
int ans(int sum,int min)
{
    // Declare variables w, x, and u as integers
    int w,x,u=1;

    // Loop through the values from min to sum/2
    for(w=min;w<sum/2;w++){
        // Calculate the remainder of the sum divided by w and store it in x
        x=sum%w;

        // Check if the remainder is equal to 0
        if(x==0){
            // Check if the value of w is less than or equal to the quotient of the sum divided by w
            if(w<=sum/w){
                // Recursively call the function ans with the new values
                u+=ans(sum/w,w);
            }
        }
    }

    // Return the result
    return u;
}
