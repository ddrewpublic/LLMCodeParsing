void factor(int a, int k) // Factorization function with a and k
{
    int b, j; // Declare integer variables b and j
    b = (int)sqrt((double)a); // Calculate the square root of a and store it in b
    for (j = k; j <= b; j++) // Loop from k to b
    {
        if (a % j == 0) // Check if a is divisible by j
        {
            sum++; // Increment the sum
            factor(a / j, j); // Recursively call factor with a/j and j
        }
    }
    j = k; // Reset j to k
}
