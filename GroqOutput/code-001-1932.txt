int sdw(int a, int b) {
    // Initialize a variable i to keep track of the counter
    int i;
    // Initialize m to 1, which will be used to store the sum of divisors
    int m = 1;
    // Loop through all numbers from a to b/2
    for (i = a; i < b / 2; i++) {
        // Check if the current number i is a divisor of b
        if (b % i == 0) {
            // Check if the divisor i is less than or equal to the quotient b/i
            if (i <= b / i) {
                // Calculate the sum of divisors of i and b/i
                m = sdw(i, b / i) + m;
            }
        }
    }
    // Return the sum of divisors m
    return m;
}

int main() {

    // Read the value of n from the user
    scanf("%d", &n);
    // Loop through all elements of the array a
    for (i = 0; i < n; i++) {

        scanf("%d", &a[i]);
    }
    // Loop through all elements of the array a
    for (i = 0; i < n; i++) {
        // Set b to the current value in the array a
        b = a[i];

        p = sdw(2, b);
        // Print the result
        printf("%d\n", p);

}
