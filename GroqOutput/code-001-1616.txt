// Function to calculate the number of ways to express a number as the product of integers
int factor(int a, int b) {
    // Initialize variables
    int i, ways;
    // Set ways to 1, as a single number itself is a way
    ways = 1;
    // If a is smaller than b, there are no ways to express a as the product of integers
    if (a < b)
        return 0;
    // Iterate from b to the square root of a
    for (i = b; i <= sqrt(a); i++) {
        // If a is divisible by i, it means i is a factor of a
        if (a % i == 0) {
            // Recursively call the function to calculate the number of ways for the remaining number
            ways += factor(a / i, i);
        }
    }
    // Return the total number of ways
    return ways;
}

// Main function to read input and print output
int main() {
    // Declare arrays to store numbers and their factors
    int n, A[99], K[99], i;
    // Read the number of test cases
    scanf("%d", &n);
    // Read each number and calculate its factors
    for (i = 0; i < n; i++) {
        scanf("%d", &A[i]);
        K[i] = factor(A[i], 2);
    }
    // Print the factors for each number
    for (i = 0; i < n; i++) {
        printf("%d\n", K[i]);
    }
    // Return 0 to indicate successful execution
    return 0;
}
