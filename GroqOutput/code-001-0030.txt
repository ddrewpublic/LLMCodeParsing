int number(int n, int low)
{
    // Declare and initialize variables 'j' and 'ans'
    int j, ans;
    // Initialize 'ans' to 0
    ans = 0;
    // Iterate through a loop starting from 'low' to the square root of 'n'
    for (j = low ; j * j<= n; j++)
    {
        // Check if 'n' is divisible by 'j'
        if (n % j == 0) 
            // If it is, recursively call 'number' with the quotient of 'n' and 'j', and add the result to 'ans'
            ans = ans + number(n / j, j);
    }
    // Add 1 to 'ans'
    ans = ans + 1;
    // Return the result 'ans'
    return ans;
}
