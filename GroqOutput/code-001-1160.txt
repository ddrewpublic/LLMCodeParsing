int fenjie(int m,int i)
{
    // Initialize an integer l to 1
    int l = 1;

    // Declare a variable j
    int j;

    // Calculate the square root of m and store it in t
    int t = sqrt((float)m);

    // Loop from j=i to j=t
    for (j = i; j <= t; ++j)
    {
        // Check if m is divisible by j
        if (m % j == 0)
            // Add the result of recursively calling fenjie with m/j and j to l
            l += fenjie(m/j, j);
    }

    // Return the value of l
    return l;
}
